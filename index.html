function renderMC() {
  const lvl = LEVELS[currentLevel];
  const q = lvl.questions[currentQuestion];
  const html = `
    <div class="game-card">
      <div class="scenario-label">${lvl.levelEmoji} ${lvl.levelTitle} — Question ${currentQuestion+1}/${lvl.questions.length}</div>

function answerMC(idx) {
  clearTimer();
  const q = LEVELS[currentLevel].questions[currentQuestion];
  const buttons = document.querySelectorAll('.choice-btn');
  buttons.forEach(b => b.classList.add('disabled'));

  if (q.choices[idx].correct) {
    buttons[idx].classList.add('correct');
    score += 10; updateHUD();
    showFeedback(true, '✅ Correct!', q.feedback.correct);
    spawnConfetti();
  } else {
    buttons[idx].classList.add('wrong');
    lives--; updateHUD();
    // show correct answer
    q.choices.forEach((c,i) => { if(c.correct) buttons[i].classList.add('correct'); });
    showFeedback(false, '❌ Not quite!', q.feedback.wrong);
    if (lives <= 0) { setTimeout(()=>endGame(), 2000); return; }
  }
  document.getElementById('btn-next').style.display = 'inline-flex';
}

function nextQuestion() {
  clearTimer(); hideFeedback();
  document.getElementById('btn-next').style.display = 'none';
  const lvl = LEVELS[currentLevel];

  if (lvl.type === 'mc') {
    if (currentQuestion + 1 < lvl.questions.length) {
      currentQuestion++; renderMC();
    } else { advanceLevel(); }
  } else if (lvl.type === 'map') {
    if (currentQuestion + 1 < lvl.questions.length) {
      currentQuestion++; renderMapQuestion();
    } else { advanceLevel(); }
  } else { advanceLevel(); }
}

function advanceLevel() {
  currentLevel++;
  if (currentLevel >= LEVELS.length) { endGame(); return; }
  currentQuestion = 0;
  showToast(`🎉 Level ${currentLevel} Complete! Moving on...`);
  setTimeout(() => renderLevel(), 1200);
}

function renderDrag() {
  const lvl = LEVELS[currentLevel];
  dragAnswers = {};
  lvl.zones.forEach(z => dragAnswers[z.id] = []);

  const html = `
    <div class="game-card">
      <div class="scenario-label">${lvl.levelEmoji} ${lvl.levelTitle}</div>
      <div class="scenario-text" style="text-align:center">${lvl.question}</div>
      <p class="drag-instructions">Drag items from the bank below into the correct category:</p>
      <div class="drag-area" id="drag-bank">
        ${lvl.items.map(item => `

               ondragstart="dragStart(event,'${item.id}')"
               ontouchstart="touchStart(event,'${item.id}')">${item.text}</div>`).join('')}
      </div>
      <div class="drop-zones">
        ${lvl.zones.map(z => `
          <div class="drop-zone" id="zone-${z.id}"
               ondragover="event.preventDefault();this.classList.add('drag-over')"
               ondragleave="this.classList.remove('drag-over')"
               ondrop="dropItem(event,'${z.id}')">
            <div class="drop-zone-label">${z.label}</div>
          </div>`).join('')}
      </div>
    </div>`;
  document.getElementById('game-content').innerHTML = html;
  document.getElementById('btn-next').style.display = 'none';
  setupTouchDrag();
}

let draggedId = null;
function dragStart(e, id) { draggedId = id; e.dataTransfer.setData('text', id); }
function dropItem(e, zoneId) {
  e.preventDefault();
  const id = draggedId || e.dataTransfer.getData('text');
  if (!id) return;
  const el = document.getElementById(id);
  const zone = document.getElementById('zone-' + zoneId);
  zone.classList.remove('drag-over');
  zone.appendChild(el);
  draggedId = null;
  checkDragCompletion();
}

function checkDragCompletion() {
  const lvl = LEVELS[currentLevel];
  const bankEl = document.getElementById('drag-bank');
  if (bankEl.querySelectorAll('.drag-item').length === 0) {
    // All placed — evaluate
    let correct = 0;
    lvl.items.forEach(item => {
      const el = document.getElementById(item.id);
      const parent = el.parentElement;
      if (parent.id === 'zone-' + item.category) { correct++; el.style.background = '#27ae60'; }
      else { el.style.background = '#e74c3c'; lives = Math.max(0, lives - 1); updateHUD(); }
    });
    score += correct * 10; updateHUD();
    const pct = Math.round((correct / lvl.items.length) * 100);
    if (correct === lvl.items.length) {
      showFeedback(true, `🎉 Perfect! ${correct}/${lvl.items.length} correct!`, 'You correctly sorted all historical events into their periods!');
      spawnConfetti();
    } else {
      showFeedback(false, `📊 ${correct}/${lvl.items.length} correct`, 'Red items were in the wrong period. Study the highlighted correct placements!');
    }
    setTimeout(() => { document.getElementById('btn-next').style.display = 'inline-flex'; }, 800);
    if (lives <= 0) setTimeout(()=>endGame(), 2200);
  }
}

// Touch drag (mobile)
let touchItemId = null, touchClone = null;
function setupTouchDrag() {
  document.querySelectorAll('.drag-item').forEach(item => {
    item.addEventListener('touchstart', touchStart, {passive:false});
    item.addEventListener('touchmove', touchMove, {passive:false});
    item.addEventListener('touchend', touchEnd);
  });
}
function touchStart(e, id) {
  e.preventDefault();
  touchItemId = id || e.currentTarget.id;
  touchClone = e.currentTarget.cloneNode(true);
  touchClone.style.cssText = `position:fixed;opacity:.7;pointer-events:none;z-index:9999;border-radius:10px;padding:9px 16px;background:#1a3c5e;color:#fff;font-weight:700;font-size:.88rem;`;
  document.body.appendChild(touchClone);
}
function touchMove(e) {
  e.preventDefault();
  if (!touchClone) return;
  const t = e.touches[0];
  touchClone.style.left = (t.clientX - 40) + 'px';
  touchClone.style.top  = (t.clientY - 20) + 'px';
}
function touchEnd(e) {
  if (!touchClone || !touchItemId) return;
  const t = e.changedTouches[0];
  touchClone.remove(); touchClone = null;
  const el = document.elementFromPoint(t.clientX, t.clientY);
  if (!el) return;
  const zone = el.closest('.drop-zone');
  if (zone) {
    const zoneId = zone.id.replace('zone-','');
    const item = document.getElementById(touchItemId);
    zone.appendChild(item);
    checkDragCompletion();
  }
  touchItemId = null;
}

// TIMER
// ════════════════════════════════════════
function startTimer(sec) {
  clearTimer(); timeLeft = sec;
  const el = document.getElementById('timer-val');
  el.textContent = timeLeft;
  el.style.color = '#fff';
  timer = setInterval(() => {
    timeLeft--;
    el.textContent = timeLeft;
    if (timeLeft <= 5) el.style.color = '#f5a623';
    if (timeLeft <= 0) {
      clearTimer();
      el.style.color = '#e74c3c';
      lives--; updateHUD();
      const lvl = LEVELS[currentLevel];
      showFeedback(false, '⏰ Time\'s Up!', 'You ran out of time. Think faster next time!');
      if (lvl.type === 'mc') {
        document.querySelectorAll('.choice-btn').forEach(b => b.classList.add('disabled'));
        const q = lvl.questions[currentQuestion];
        document.querySelectorAll('.choice-btn').forEach((b,i) => { if(q.choices[i].correct) b.classList.add('correct'); });
      }
      document.getElementById('btn-next').style.display = 'inline-flex';
      if (lives <= 0) setTimeout(()=>endGame(), 2000);
    }
  }, 1000);
}

function clearTimer() { if (timer) { clearInterval(timer); timer = null; } }

function endGame() {
  clearTimer();
  let grade, medal, msg;
  const pct = score;
  if (pct >= 90) { grade='A+'; medal='🥇'; msg='Outstanding! You are a true civic champion!'; }
  else if(pct >= 80) { grade='A';  medal='🥇'; msg='Excellent! You have strong Social Studies knowledge!'; }
  else if(pct >= 70) { grade='B';  medal='🥈'; msg='Good job! Keep practicing your civic and history skills!'; }
  else if(pct >= 60) { grade='C';  medal='🥉'; msg='Fair effort! Review your notes and try again!'; }
  else               { grade='D';  medal='🏅'; msg='Keep going! Every champion starts somewhere. Try again!'; }

  document.getElementById('final-score').textContent = score;
  document.getElementById('grade-badge').textContent = 'Grade ' + grade;
  document.getElementById('end-medal').textContent = medal;
  document.getElementById('end-message').textContent = msg;
  showScreen('screen-end');
  if(pct >= 70) spawnConfetti(40);
}

function spawnConfetti(n = 20) {
  const colors = ['#f5a623','#27ae60','#2980b9','#e74c3c','#9b59b6','#1abc9c'];
  for (let i = 0; i < n; i++) {
    const el = document.createElement('div');
    el.className = 'confetti-piece';
    el.style.cssText = `
      left:${Math.random()*100}vw;
      background:${colors[Math.floor(Math.random()*colors.length)]};
      width:${6+Math.random()*8}px;
      height:${6+Math.random()*8}px;
      animation-duration:${1.5+Math.random()*2}s;
      animation-delay:${Math.random()*.5}s;
      border-radius:${Math.random() > .5 ? '50%' : '2px'};
    `;
    document.body.appendChild(el);
    el.addEventListener('animationend', () => el.remove());
  }
}
